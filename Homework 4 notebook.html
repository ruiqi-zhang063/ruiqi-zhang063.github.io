Description¶

This homework uses table 15.

Firstly, it uses openppyxl to obtain the indent in excel, then based on the indent to divide the hierarchical relationship in index.

The dashboard has three figures, one drop-down box and one time slider. The drop-down can choose male data, female data or total data,while the time slider can choose year. The first bar figure can show the doctorate recipents number by broad field this year. When we move the cursor to the histogram of a certain field, the second bar figure will show the subdivisions of doctorate recipents in this field, and the third line chart will show this field doctorate recipents every year.

Run the dashboard¶

In [ ]:
external_stylesheets = ['https://codepen.io/chriddyp/pen/bWLwgP.css']

app = dash.Dash(__name__, external_stylesheets=external_stylesheets)

available_indicators = ['Total','Male','Female']

app.layout = html.Div([
    
    html.H1(
        children='Doctorate Recipients by Sex and Major Field',
        style={
            'textAlign': 'center'
        }
    ),
    
    html.Div([
        html.Div([
            dcc.Dropdown(
                id='Gender',
                options=[{'label': i, 'value': i} for i in available_indicators],
                value='Total'
            )
        ],
        style={'width': '49%', 'display': 'inline-block'}),
    ], style={
        'padding': '10px 5px'
    }),

    html.Div([
        'Doctorate Recipients by Broad Field',
        dcc.Graph(
            id='Broad-Field-Bar',
            hoverData={'points': [{'customdata': 'Engineering'}]}
        )
    ], style={'width': '49%', 'display': 'inline-block', 'padding': '0 20'}),
    
    html.Div([
        'Doctorate Recipients in Specific Field',
        dcc.Graph(id='Specific-Field-Bar'),
    ], style={'display': 'inline-block', 'width': '49%'}),
    
    html.Div(dcc.Slider(
        id='Year-Slider',
        min=2008,
        max=2017,
        value=2017,
        marks={str(year): str(year) for year in range(2008, 2018)},
        step=None
    ), style={'width': '49%', 'padding': '0px 20px 20px 20px'}),
    
    html.Div([
        'Doctorate Recipients Every Year in Broad Field',
        dcc.Graph(id='Time-Series'),
    ]),
])

@app.callback(
    dash.dependencies.Output('Broad-Field-Bar', 'figure'),
    [dash.dependencies.Input('Gender', 'value'),
     dash.dependencies.Input('Year-Slider', 'value')
    ])
def update_graph(Gender, Year):
    if Gender == 'Total':
        dff = data_indent_total[data_indent_total['indent'] == 1]
    elif Gender == 'Male':
        dff = data_indent_male[data_indent_male['indent'] == 2]
    else:
        dff = data_indent_female[data_indent_female['indent'] == 2]

    fig = px.bar(dff, x = 'Sex and major field of study', y = str(Year))

    fig.update_traces(customdata = dff['Sex and major field of study'])
    
    fig.update_xaxes(title = 'Broad Field')
    
    fig.update_yaxes(title='Doctorate Recipients Number')

    fig.update_layout(margin={'l': 40, 'b': 40, 't': 10, 'r': 0}, hovermode='closest')

    return fig

@app.callback(
    dash.dependencies.Output('Specific-Field-Bar', 'figure'),
    [dash.dependencies.Input('Broad-Field-Bar', 'hoverData'),
     dash.dependencies.Input('Gender', 'value'),
     dash.dependencies.Input('Year-Slider', 'value')
    ])
def update_Specific_Field_Bar(hoverData, Gender, Year):
    if Gender == 'Total':
        dff = data_indent_total[data_indent_total['Broad Field'] == hoverData['points'][0]['customdata']]
    elif Gender == 'Male':
        dff = data_indent_male[data_indent_male['Broad Field'] == hoverData['points'][0]['customdata']]
    else:
        dff = data_indent_female[data_indent_female['Broad Field'] == hoverData['points'][0]['customdata']]
    
    fig = px.bar(dff, x = 'Sex and major field of study', y = str(Year))
    
    fig.update_xaxes(title = 'Specific Field in ' + hoverData['points'][0]['customdata']) 
    
    fig.update_yaxes(title='Doctorate Recipients Number')

    return fig

@app.callback(
    dash.dependencies.Output('Time-Series', 'figure'),
    [dash.dependencies.Input('Broad-Field-Bar', 'hoverData'),
     dash.dependencies.Input('Gender', 'value'),
    ])

def create_time_series(hoverData, Gender):
    
    if Gender == 'Total':
        dff = data_indent_total[data_indent_total['indent'] == 1]
    elif Gender == 'Male':
        dff = data_indent_male[data_indent_male['indent'] == 2]
    else:
        dff = data_indent_female[data_indent_female['indent'] == 2]
        
    dff = dff[dff['Sex and major field of study'] == hoverData['points'][0]['customdata']]
    
    x = range(2008,2018)
    
    y = list(dff.iloc[0])[3:-1]

    fig = px.scatter(x = x, y = y)

    fig.update_traces(mode='lines+markers')

    fig.update_xaxes(showgrid=False)
    
    fig.update_xaxes(title = 'Year')
    
    fig.update_yaxes(title='Doctorate Recipients Number')
    
    fig.add_annotation(x=0, y=0.85, xanchor='left', yanchor='bottom',
                       xref='paper', yref='paper', showarrow=False, align='left', 
                       text = 'Doctorate Recipients Number in ' + hoverData['points'][0]['customdata'])

    return fig

if __name__ == '__main__':
    app.run_server()
